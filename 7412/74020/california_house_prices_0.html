California House Prices

The california house prices dataset is published from the US Census Bureau. It contains variables such as population, median income, and median housing price. The objective is to build a model that will predict a median housing price given the district and all other variables.

Each row represents a "block", which is a plot of land as measured in the *Lot and Block Survey System* in the United States and Canada.

def frequency(df, col):
    return_dict = {}
    return_dict[col] = df[col].unique().tolist()
    return_dict[col].sort()
    return_dict['count'] = []
    for item in return_dict[col]:
        return_dict['count'].append(len(df[df[col] == item]))

    return_df = pd.DataFrame(return_dict)
    return return_df

def crossFrequency(df, col1, col2, dfCondition=None):
    return_df = None
    return_dict = {}
    return_dict[col1] = df[col1].unique().tolist()
    return_dict[col1].sort()

    sortedCol2 = df[col2].unique().tolist()
    sortedCol2.sort()

    for item in sortedCol2:
        return_dict[str(item)] = []

    for i in sortedCol2:
        for j in return_dict[col1]:

            if dfCondition:
                return_dict[str(i)].append(len(eval("df[(df[col1] == j) & (df[col2] == i) & " + dfCondition + "]")))
            else:
                return_dict[str(i)].append(len(df[(df[col1] == j) & (df[col2] == i)]))

            #return_dict[str(i)].append(len(df[(df[col1] == j) & (df[col2] == i)]))

    return_df = pd.DataFrame(return_dict)
    return return_df



def getFrequencies(df, col1, col2, col2Val):
    return_df = None
    return_df_col2_label = col2 +"_num_eq_" + str(col2Val)
    return_dict = {}
    return_dict[col1] = df[col1].unique().tolist()
    return_dict[col1].sort()
    return_dict[return_df_col2_label] = []
    for item in return_dict[col1]:
        return_dict[return_df_col2_label].append(len(df[(df[col1] == item) & (df[col2] == col2Val)]))

    return_df = pd.DataFrame(return_dict)
    return return_df


def summarizeFrequencies(df, x_col, y_col):
    temp_df = getFrequencies(df, x_col, y_col, 1)
    return_dict = {}
    return_dict[x_col] = temp_df[x_col].tolist()
    y_name = y_col + "_num_eq_1"
    y_mean = y_col + "_mean"
    return_dict['Freq'] = frequency(df, x_col)['count'].tolist()
    return_dict[y_col] = temp_df[y_name].tolist()
    return_dict[y_mean] = []

    for i,item in enumerate(return_dict[y_col]):
        avg = return_dict[y_col][i]/return_dict['Freq'][i]
        return_dict[y_mean].append(avg)

    return_df = pd.DataFrame(return_dict)

    return return_df

import statsmodels.api as smf

def forward_selected(data, response):
    """Linear model designed by forward selection.

    Parameters:
    -----------
    data : pandas DataFrame with all possible predictors and response

    response: string, name of response column in data

    Returns:
    --------
    model: an "optimal" fitted statsmodels linear model
           with an intercept
           selected by forward selection
           evaluated by adjusted R-squared
    """
    remaining = set(data.columns)
    remaining.remove(response)
    selected = []
    current_score, best_new_score = 0.0, 0.0
    while remaining and (current_score == best_new_score):
        scores_with_candidates = []
        for candidate in remaining:
            #formula = "{} ~ {} + 1".format(response,
            #                               ' + '.join(selected + [candidate]))
            #score = smf.ols(formula, data).fit().rsquared_adj
            x = df[candidate]
            y = df[response]
            score = smf.OLS(y, x).fit().rsquared
            scores_with_candidates.append((score, candidate))
        scores_with_candidates.sort()
        best_new_score, best_candidate = scores_with_candidates.pop()
        if current_score < best_new_score:
            remaining.remove(best_candidate)
            selected.append(best_candidate)
            current_score = best_new_score
    '''
    formula = "{} ~ {} + 1".format(response,
                                   ' + '.join(selected))
    '''
    x = df[candidate]
    y = df[response]
    model = smf.OLS(y, x).fit()
    #model = smf.ols(formula, data).fit()
    return model
